Def LEVEL(NAME, D1, D2, W)
  {
    type = EnemyZLevel {
      enemy = NAME
      attackChance = 1.0
    }
    minDepth = D1
    maxDepth = D2
    weight = W
  }
End

Def LEVEL(NAME, D1, D2) LEVEL(NAME, D1, D2, 1.0) End
